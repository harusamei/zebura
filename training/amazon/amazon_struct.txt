Table: product
Purpose: The product database stores product information,price, discount, rating, and so on.
Columns:
    product_id, varchar, PRIMARY KEY, COMMENT( ID; item_id )
    product_name, varchar, Language(English), COMMENT( name;  title )
    category, varchar, Language(English), COMMENT( type;  genre )
    discounted_price, float, COMMENT( price_after_discount;  sale_price )
    actual_price, float, COMMENT( original_price;  list_price )
    discount_percentage, float, COMMENT( discount_percent;  savings_percent )
    rating, float, COMMENT( user_rating;  average_rating )
    rating_count, int, COMMENT( review_count;  feedback_count )
    about_product, varchar, Language(English), COMMENT( product_description;  details )
    user_id, varchar, COMMENT( reviewer_id;  customer_id )
    user_name, varchar, COMMENT( reviewer_name;  customer_name )
    review_id, varchar, COMMENT( comment_id;  feedback_id )                        
    review_title, varchar, Language(English), COMMENT( comment_title;  review_subject )
    review_content, text, Language(English), COMMENT( comment_content;  review_text )
    img_link, varchar, COMMENT( image_url;  picture_link )
    product_link, varchar, COMMENT( buy_link;  purchase_url )
    brand, varchar, Language(English), COMMENT( manufacturer;  brand_name )
    compatible, varchar, COMMENT( compatible_with;  fits_with )
    scenario, varchar, COMMENT( use_case;  application )
